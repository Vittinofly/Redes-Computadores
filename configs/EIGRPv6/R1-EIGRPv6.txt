hostname R1

interface FastEthernet0/0

 ipv6 enable
 ipv6 address 2001:DB8:CAFE:1::1/64

 ipv6 ospf 1 area 0

 ipv6 ospf 1 area 0
 no shutdown

ipv6 unicast-routing
ipv6 router eigrp 10
 no shutdown
